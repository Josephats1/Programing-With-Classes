‚úÖ Program.cs ‚Äî Main Program File

using System;
using System.Collections.Generic;
using System.IO;

class Program
{
    static List<Product> products = new List<Product>();

    static void Main(string[] args)
    {
        Console.WriteLine("Welcome to the Grocery Inventory System");

        LoadProductsFromFile("products.txt");
        DisplayMenu();
    }

    static void LoadProductsFromFile(string fileName)
    {
        try
        {
            string[] lines = File.ReadAllLines(fileName);
            foreach (var line in lines)
            {
                string[] parts = line.Split(',');
                string name = parts[0];
                double price = double.Parse(parts[1]);

                Product product = new Product(name, price);
                products.Add(product);
            }

            Console.WriteLine($"Loaded {products.Count} products from file.\n");
        }
        catch (Exception ex)
        {
            Console.WriteLine("Error reading file: " + ex.Message);
        }
    }

    static void DisplayMenu()
    {
        string input = "";
        while (input != "4")
        {
            Console.WriteLine("1. View Products");
            Console.WriteLine("2. Add Product");
            Console.WriteLine("3. Save Products");
            Console.WriteLine("4. Exit");
            Console.Write("Choose an option: ");
            input = Console.ReadLine();

            switch (input)
            {
                case "1":
                    ViewProducts();
                    break;
                case "2":
                    AddProduct();
                    break;
                case "3":
                    SaveProductsToFile("products.txt");
                    break;
                case "4":
                    Console.WriteLine("Exiting...");
                    break;
                default:
                    Console.WriteLine("Invalid option. Try again.\n");
                    break;
            }
        }
    }

    static void ViewProducts()
    {
        Console.WriteLine("\n=== Product List ===");
        foreach (Product product in products)
        {
            product.PrintInfo();
        }
        Console.WriteLine();
    }

    static void AddProduct()
    {
        Console.Write("Enter product name: ");
        string name = Console.ReadLine();

        Console.Write("Enter product price: ");
        double price = double.Parse(Console.ReadLine());

        Product newProduct = new Product(name, price);
        products.Add(newProduct);

        Console.WriteLine("Product added!\n");
    }

    static void SaveProductsToFile(string fileName)
    {
        try
        {
            using (StreamWriter writer = new StreamWriter(fileName))
            {
                foreach (Product product in products)
                {
                    writer.WriteLine($"{product.Name},{product.Price}");
                }
            }

            Console.WriteLine("Products saved successfully.\n");
        }
        catch (Exception ex)
        {
            Console.WriteLine("Error saving file: " + ex.Message);
        }
    }
}
